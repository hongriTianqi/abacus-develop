295  if(pv.in_this_processor(iwt0, iwt1))
296  Hk_seq_tmp[iwt0][iwt1] = (1-mixing_beta) * Hk_seq[Hk_seq.size()-1][iwt0][iwt1] + mixing_beta * Hk_new[iwt0][iwt1];
297  Hk_seq.emplace_back(Hk_seq_tmp);

299  Hk_seq.pop_front();
300 
301  auto pos_mod = [](const int i, const int N){ return (i%N+N)%N; };

303  {
304  const int alpha_size = Hk_seq.size()-1;
305  const int alpha_begin = GlobalC::CHR_MIX.get_idstep() - alpha_size;
306  return pos_mod( alpha_begin+i, GlobalC::CHR_MIX.get_dstep() );
307  };
308  for(size_t iwt0=0; iwt0!=GlobalV::NLOCAL; ++iwt0)
309  for(size_t iwt1=0; iwt1!=GlobalV::NLOCAL; ++iwt1)
310  if(pv.in_this_processor(iwt0, iwt1))
311  {
312  Hk[iwt0][iwt1] = (1+GlobalC::CHR_MIX.get_alpha()[index(Hk_seq.size()-2)]) * Hk_seq[Hk_seq.size()-1][iwt0][iwt1];
313  for( int i=1; i<=Hk_seq.size()-2; ++i )
314  Hk[iwt0][iwt1] += ( GlobalC::CHR_MIX.get_alpha()[index(i-1)] - GlobalC::CHR_MIX.get_alpha()[index(i)] ) * Hk_seq[i][iwt0][iwt1];
315  Hk[iwt0][iwt1] -= GlobalC::CHR_MIX.get_alpha()[index(0)] * Hk_seq[0][iwt0][iwt1];
316  }
317  }
