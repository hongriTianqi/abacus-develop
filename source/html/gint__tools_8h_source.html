291  template<typename T>
292  Array_Pool<T>::Array_Pool(const int nr, const int nc) // Attention: uninitialized
293  {
294  ptr_1D = new T[nr*nc];
295  ptr_2D = new T*[nr];
296  for (int ir=0; ir<nr; ++ir)
297  ptr_2D[ir] = &ptr_1D[ir*nc];
298  }
299 
300  template<typename T>
301  Array_Pool<T>::Array_Pool(Array_Pool<T> &&array)
302  {
303  ptr_1D = array.ptr_1D;
304  ptr_2D = array.ptr_2D;
305  delete[] array.ptr_2D;
306  delete[] array.ptr_1D;
307  }
308 
309  template<typename T>

311  {
