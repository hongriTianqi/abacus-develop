70 
71  int SENDPROW = 0, SENDPCOL = 0, tag = 0;
72  T *tmp = new T[nblk];
73 
74  // do iteration for matrix, distribute old_matrix to each process, pass a block each time
75  for (int row = 0; row < nFull; row++)
76  {
77  int recv_prow = (row / nblk) % nprows; // the row number of recive process
78  int nm_row = ((row / nblk) / nprows) * nblk + row % nblk; // row number of block in new_matrix
79  for (int col = 0; col < nFull; col += nblk)
80  {
81  int recv_pcol = (col / nblk) % npcols; // the column number of recive process
82  int nm_col = ((col / nblk) / npcols) * nblk + col % nblk;
83  int pass_length = std::min(nFull - col, nblk); // nFull may not be devided by nblk;
84 
85  if (myprow == SENDPROW && mypcol == SENDPCOL)
86  {
87  if (myprow == recv_prow && mypcol == recv_pcol)
88  {
89  for (int i = 0; i < pass_length; i++)
90  {
91  new_matrix[(nm_col + i) * na_rows + nm_row] = hmatrix[nFull * row + col + i];
92  }
93  }
94  else
95  {

97  MPI_Send(&(hmatrix[nFull * row + col]),

108  tag,
109  MPI_COMM_WORLD);
110  }
111  }
112  else if (myprow == recv_prow && mypcol == recv_pcol)
113  {
