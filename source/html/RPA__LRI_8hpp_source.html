29 template <typename Tdata> void RPA_LRI<Tdata>::cal_rpa_cv()
30 {
31  std::vector<TA> atoms(GlobalC::ucell.nat);
32  for (int iat = 0; iat < GlobalC::ucell.nat; ++iat)
33  atoms[iat] = iat;
34  const std::array<Tcell, Ndim> period = {GlobalC::kv.nmp[0], GlobalC::kv.nmp[1], GlobalC::kv.nmp[2]};
35 
36  const std::array<Tcell, Ndim> period_Vs = LRI_CV_Tools::cal_latvec_range<Tcell>(1 + this->info.ccp_rmesh_times);
37  const std::pair<std::vector<TA>, std::vector<std::vector<std::pair<TA, std::array<Tcell, Ndim>>>>> list_As_Vs
38  = RI::Distribute_Equally::distribute_atoms(this->mpi_comm, atoms, period_Vs, 2, false);
39 
40  std::map<TA, std::map<TAC, RI::Tensor<Tdata>>> Vs
41  = exx_lri_rpa.cv.cal_Vs(list_As_Vs.first, list_As_Vs.second[0], {{"writable_Vws", true}});
42  this->Vs_period = RI::RI_Tools::cal_period(Vs, period);
43 
44  const std::array<Tcell, Ndim> period_Cs = LRI_CV_Tools::cal_latvec_range<Tcell>(2);
45  const std::pair<std::vector<TA>, std::vector<std::vector<std::pair<TA, std::array<Tcell, Ndim>>>>> list_As_Cs
46  = RI::Distribute_Equally::distribute_atoms_periods(this->mpi_comm, atoms, period_Cs, 2, false);
47 
48  std::pair<std::map<TA, std::map<TAC, RI::Tensor<Tdata>>>,
49  std::array<std::map<TA, std::map<TAC, RI::Tensor<Tdata>>>, 3>>
50  Cs_dCs = exx_lri_rpa.cv.cal_Cs_dCs(list_As_Cs.first,
51  list_As_Cs.second[0],
52  {{"cal_dC", false},
53  {"writable_Cws", true},
54  {"writable_dCws", true},
55  {"writable_Vws", false},
56  {"writable_dVws", false}});
57  std::map<TA, std::map<TAC, RI::Tensor<Tdata>>> &Cs = std::get<0>(Cs_dCs);
58  this->Cs_period = RI::RI_Tools::cal_period(Cs, period);
59 }

251 template <typename Tdata> void RPA_LRI<Tdata>::out_coulomb_k()
252 {
253  int all_mu = 0;
254  vector<int> mu_shift(GlobalC::ucell.nat);
255  for (int I = 0; I != GlobalC::ucell.nat; I++)
256  {
257  mu_shift[I] = all_mu;
258  all_mu += exx_lri_rpa.cv.get_index_abfs_size(GlobalC::ucell.iat2it[I]);
259  }
260  const int nks_tot = GlobalV::NSPIN == 2 ? (int)GlobalC::kv.nks / 2 : GlobalC::kv.nks;
261  std::stringstream ss;
262  ss << "coulomb_mat_" << GlobalV::MY_RANK << ".txt";
263 
264  std::ofstream ofs;
265  ofs.open(ss.str().c_str(), std::ios::out);
266 
267  ofs << nks_tot << std::endl;
268  for (auto &Ip: this->Vs_period)
269  {
270  auto I = Ip.first;
271  size_t mu_num = exx_lri_rpa.cv.get_index_abfs_size(GlobalC::ucell.iat2it[I]);
272 
273  for (int ik = 0; ik != nks_tot; ik++)
274  {
275  map<size_t, RI::Tensor<complex<double>>> Vq_k_IJ;
276  for (auto &JPp: Ip.second)
277  {
278  auto J = JPp.first.first;
279 
280  auto R = JPp.first.second;
281  if (J < I)
282  continue;
283  RI::Tensor<complex<double>> tmp_VR = RI::Global_Func::convert<complex<double>>(JPp.second);
284 
285  const double arg = 1
286  * (GlobalC::kv.kvec_c[ik] * (RI_Util::array3_to_Vector3(R) * GlobalC::ucell.latvec))
287  * ModuleBase::TWO_PI; // latvec
288  const complex<double> kphase = complex<double>(cos(arg), sin(arg));
289  if (Vq_k_IJ[J].empty())
290  Vq_k_IJ[J] = RI::Tensor<complex<double>>({tmp_VR.shape[0], tmp_VR.shape[1]});
291  Vq_k_IJ[J] = Vq_k_IJ[J] + tmp_VR * kphase;
292  }
293  for (auto &vq_Jp: Vq_k_IJ)
294  {
295  auto iJ = vq_Jp.first;
296  auto &vq_J = vq_Jp.second;
297  size_t nu_num = exx_lri_rpa.cv.get_index_abfs_size(GlobalC::ucell.iat2it[iJ]);
298  ofs << all_mu << " " << mu_shift[I] + 1 << " " << mu_shift[I] + mu_num << " " << mu_shift[iJ] + 1
299  << " " << mu_shift[iJ] + nu_num << std::endl;
300  ofs << ik + 1 << " " << GlobalC::kv.wk[ik] / 2.0 * GlobalV::NSPIN << std::endl;
301  for (int i = 0; i != vq_J.data->size(); i++)
302  {
303  ofs << std::setw(21) << std::fixed << std::setprecision(12) << (*vq_J.data)[i].real()
304  << std::setw(21) << std::fixed << std::setprecision(12) << (*vq_J.data)[i].imag() << std::endl;
305  }
306  }
307  }
308  }
309  ofs.close();
310 }
